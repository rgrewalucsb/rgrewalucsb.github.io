Looping

	var i = 1;
	while (i < 0){
		console.log(i);
	};

	//Looping from 10 to 1
	for (var i = 10; i > 0;i --){
		console.log(i);
	};

	if (*some condition is true*) {

	};
	else if (*some OTHER condition is true*) {
